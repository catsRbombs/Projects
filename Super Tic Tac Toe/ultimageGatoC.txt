   144	void createButtonGrid()
   145	{
   146	    GtkWidget *boton;
   147	    struct Coordenada *coord;
   148	
   149	    // Posiciones donde se van a ir poniendo los botónes
   150	    int xInicial = 8;
   151	    int yInicial = 3;
   152	    int x = xInicial;
   153	    int y = yInicial;
   154	
   155	    for (int i = 0; i < 9; i++)
   156	    {
   157	
   158	        // Cuando cambian de fila hay que cambiar el Y
   159	        if (i == 3)
   160	        {
   161	            xInicial = x = 8;
   162	            yInicial = y += 212;
   163	        }
   164	        else if (i == 6)
   165	        {
   166	            xInicial = x = 8;
   167	            yInicial = y += 208;
   168	        }
   169	
   170	        for (int j = 0; j < 3; j++)
   171	        {
   172	            for (int k = 0; k < 3; k++)
   173	            {
   174	
   175	                boton = gtk_button_new_with_label("");
   176	                gtk_widget_set_size_request(boton, 50, 50);
   177	                // TODO: Poner comandos del juego
   178	
   179	                // Creo la coordenada para asignarle al boton en el comando
   180	                coord = (struct Coordenada *)malloc(sizeof(struct Coordenada));
   181	                coord->x = i;
   182	                coord->y = j;
   183	                coord->z = k;
   184	
   185	                grid[i][j][k] = boton;
   186	                g_signal_connect(grid[i][j][k], "clicked", G_CALLBACK(cellClicked), coord);
   187	                gtk_fixed_put(GTK_FIXED(fixed), grid[i][j][k], x, y);
   188	
   189	                x += 60;
   190	            }
   191	            x = xInicial;
   192	            y += 60;
   193	        }
   194	        y = yInicial;
   195	        xInicial = x += 207;
   196	    }
   197	}

   230	void cellClicked(GtkWidget *widget, gpointer data)
   231	{   
   232	    
   233	    gtk_widget_set_sensitive(GTK_WIDGET(cambiarTokenBtn), FALSE);
   234	    struct Coordenada *coord = (struct Coordenada *)data;
   235	
   236	    // TODO: Implementar lógica del gato si es válido poner la casilla ahí
   237	    if (is_valid_move(coord->x, coord->y, coord->z))
   238	    {
   239	
   240	        GtkWidget *boton = grid[coord->x][coord->y][coord->z];
   241	
   242	        printf("Botón en la casilla (%d, %d, %d) presionado\n", coord->x, coord->y, coord->z);
   243	
   244	        // Cambia el ícono del botón a la O o la X dependiendo del turno del jugador
   245	        changeCellIcon(boton);
   246	        // Le cambia el color al botón a uno gris
   247	        highlightCell(coord);
   248	
   249	        // Coloca la X o la O en la matriz (datos)
   250	        make_move(coord->x, coord->y, coord->z);
   251	
   252	        // Verificar si se hizo una línea de 3
   253	        if (check_board_victory(coord->x))
   254	        {
   255	            changeBigCellIcon(coord->x);
   256	        }
   257	        else
   258	        {
   259	            // Si no hay una línea de 3, verifica si ese subtablero se llenó
   260	            check_board_tie(coord->x);
   261	        }
   262	
   263	        // Verifica si el jugador ganó el juego
   264	        if (check_game_victory())
   265	        {
   266	            victory();
   267	            return;
   268	        }
   269	
   270	        switchTurno();
   271	
   272	
   273	        if (botPlay)
   274	        {
   275	            turnoBot();
   276	        }
   277	    }
   278	    else{
   279	        printf("Botón en la casilla (%d, %d, %d) inavlida!\n", coord->x, coord->y, coord->z);
   280	    }
   281	}

   324	void changeCellIcon(GtkWidget *boton)
   325	{
   326	    GtkWidget *imagen;
   327	
   328	    if (turno == 0)
   329	    { // Si es turno de O
   330	        imagen = gtk_image_new_from_file(O_SMALL);
   331	    }
   332	    else
   333	    { // Si es turno de X
   334	        imagen = gtk_image_new_from_file(X_SMALL);
   335	    }
   336	
   337	    if (imagen == NULL)
   338	    {
   339	        printf("Error al cargar la imagen.\n");
   340	    }
   341	
   342	    gtk_button_set_image(GTK_BUTTON(boton), imagen);
   343	    gtk_widget_queue_draw(GTK_WIDGET(boton));
   344	    gtk_widget_show(imagen); // Asegurarse de que la imagen sea visible
   345	}

   347	void changeButtonColor(GtkWidget *boton, const char *css)
   348	{
   349	    GtkCssProvider *provider = gtk_css_provider_new();
   350	    GtkStyleContext *context = gtk_widget_get_style_context(boton);
   351	
   352	    gtk_css_provider_load_from_data(provider, css, -1, NULL);
   353	    gtk_style_context_add_provider(context, GTK_STYLE_PROVIDER(provider), GTK_STYLE_PROVIDER_PRIORITY_USER);
   354	
   355	    g_object_unref(provider);
   356	}

   358	void highlightCell(struct Coordenada *coord)
   359	{
   360	    int x = coord->x;
   361	    int y = coord->y;
   362	    int z = coord->z;
   363	
   364	    for (int i = 0; i < 9; i++)
   365	    {
   366	        for (int j = 0; j < 3; j++)
   367	        {
   368	            for (int k = 0; k < 3; k++)
   369	            {
   370	                GtkWidget *boton = grid[i][j][k];
   371	                if (i == x && j == y && k == z)
   372	                {
   373	                    // Le cambia el color del botón a uno gris
   374	                    changeButtonColor(boton, css_highlight);
   375	                }
   376	                else
   377	                {
   378	                    // Restablece el color predeterminado del boton
   379	                    changeButtonColor(boton, css_default);
   380	                }
   381	            }
   382	        }
   383	    }
   384	}

   477	void reiniciarPartida(){
   478	    //Settea el turno inicial a O
   479	    turno = 0;
   480	    gtk_image_set_from_file(GTK_IMAGE(turnoJugadorImg), O_BIG);
   481	
   482	    //Limpia el tablero de matrices en el ensamblador
   483	    restart_game();
   484	
   485	    //Borra las imágenes y rehabilita los botones
   486	    for(int i=0; i<9; i++){
   487	        for(int j=0; j<3; j++){
   488	            for(int k=0; k<3; k++){
   489	                gtk_button_set_image(GTK_BUTTON(grid[i][j][k]), NULL);
   490	                gtk_widget_set_sensitive(GTK_WIDGET(grid[i][j][k]), TRUE); 
   491	                changeButtonColor(grid[i][j][k], css_default);
   492	            }
   493	        }
   494	    }
   495	
   496	    
   497	    for(int j=0; j<3; j++){
   498	        for(int k=0; k<3; k++){
   499	            gtk_image_set_from_file(GTK_IMAGE(megaGrid[j][k]), NULL);
   500	        }
   501	    }
   502	
   503	    gtk_widget_set_sensitive(GTK_WIDGET(cambiarTokenBtn), TRUE);
   504	
   505	    switchTurno();
   506	}

   526	void victory()
   527	{
   528	// Crear un diálogo para mostrar quién ganó
   529	    GtkWidget *dialog = gtk_dialog_new_with_buttons(
   530	        "",
   531	        GTK_WINDOW(window),
   532	        GTK_DIALOG_MODAL | GTK_DIALOG_DESTROY_WITH_PARENT,
   533	        "_OK",
   534	        GTK_RESPONSE_OK,
   535	        NULL);
   536	
   537	    // Crear un contenedor para los widgets
   538	    GtkWidget *content_area = gtk_dialog_get_content_area(GTK_DIALOG(dialog));
   539	    GtkWidget *vbox = gtk_box_new(GTK_ORIENTATION_VERTICAL, 10);
   540	    gtk_box_set_homogeneous(GTK_BOX(vbox), FALSE);
   541	    gtk_container_add(GTK_CONTAINER(content_area), vbox);
   542	
   543	    // Incrementar el puntaje y establecer imagen del ganador
   544	    GtkWidget *imagen = gtk_image_new();
   545	    if (turno == 0)
   546	    {   
   547	        pointP1++;
   548	        gtk_image_set_from_file(GTK_IMAGE(imagen), O_BIG);
   549	    }
   550	    else
   551	    {
   552	        pointP2++;
   553	        gtk_image_set_from_file(GTK_IMAGE(imagen), X_BIG);
   554	    }
   555	
   556	    // Crear el label del ganador
   557	    GtkWidget *label_title = gtk_label_new("¡GANADOR!");
   558	    gtk_box_pack_start(GTK_BOX(vbox), label_title, TRUE, TRUE, 10);
   559	
   560	    // Crear el label con las estadísticas de puntos
   561	    char stats[50];
   562	    snprintf(stats, sizeof(stats), "P1 Wins: %d | P2 Wins: %d", pointP1, pointP2);
   563	    GtkWidget *label_stats = gtk_label_new(stats);
   564	    gtk_box_pack_start(GTK_BOX(vbox), label_stats, TRUE, TRUE, 10);
   565	
   566	    // Centrar la imagen del ganador
   567	    gtk_widget_set_halign(imagen, GTK_ALIGN_CENTER);
   568	    gtk_box_pack_start(GTK_BOX(vbox), imagen, TRUE, TRUE, 10);
   569	
   570	    // Mostrar el dialog
   571	    gtk_widget_show_all(dialog);
   572	
   573	    // Esperar la confirmación del usuario y cerrar el dialog
   574	    gtk_dialog_run(GTK_DIALOG(dialog));
   575	    gtk_widget_destroy(dialog);
   576	
   577	    // Deshabilitar todos los botones para que no puedan seguir jugando hasta que reinicien
   578	    for (int i = 0; i < 9; i++)
   579	    {
